47 if self.run_against == "dataframe":

49 raise SDCDataValidationSetupError( 

50 f"{self.validation_test_name} test is running against an SDCDataframe, but, " 

51 f"sdc_dataframe_ was not passed in.") 

52 elif not isinstance(self.sdc_dataframe, pd.core.frame.DataFrame): 

53 raise SDCDataValidationSetupError( 

54 f"{self.validation_test_name} test is running against an SDCDataframe, but, "

57 elif self.run_against == "database": 

58 # Ensure database_info_ was passed in.

61 f"but, database_info_ was not passed in.") 

62 # Ensure that the minimum necessary keys exist in database_info_ and have values.
